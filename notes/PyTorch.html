# buildinv custom network
class NN_Regression(nn.Module):
    super(NN_regression, self).__init():
    #initialize components
        self.layer1 = nn.Linear(3,6)
        self.layer2 = nn.Linear(4,1)
        self.relu = nn.ReLU()
    def forward(self, x):
        x = self.layer1(x)
        x = self.relu(x)
        x = self.layer2(x)
        return x
